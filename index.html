üîä PS5 Audio Loopback Script 
#!/bin/bash
SOURCE="alsa_input.usb-0d8c_USB_Sound_Device-00.analog-stereo"
SINK="alsa_output.usb-Kingston_HyperX_Cloud_II_Wireless_000000000001-00.analog-stereo"
STATE_FILE="$HOME/.dp1_state"
LOOPBACK_FILE="$HOME/.loopback_module_id"

if [ -f "$STATE_FILE" ]; then STATE=$(cat "$STATE_FILE"); else STATE="enabled"; fi

if [ "$STATE" = "enabled" ]; then
    kscreen-doctor output.DP-1.disable
    echo "disabled" > "$STATE_FILE"
    if [ ! -f "$LOOPBACK_FILE" ]; then
        MODULE_ID=$(pactl load-module module-loopback source="$SOURCE" sink="$SINK" latency_msec=10)
        echo "$MODULE_ID" > "$LOOPBACK_FILE"
    fi
    notify-send -i video-display "DP-1 desligado" "Audio loopback ON"
else
    kscreen-doctor output.DP-1.enable output.DP-1.primary
    echo "enabled" > "$STATE_FILE"
    if [ -f "$LOOPBACK_FILE" ]; then
        pactl unload-module "$(cat "$LOOPBACK_FILE")"
        rm -f "$LOOPBACK_FILE"
    fi
    notify-send -i video-display "DP-1 ligado" "Audio loopback OFF"
fi
